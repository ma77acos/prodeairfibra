# Compilar el proyecto
./gradlew build

# Ejecutar la aplicación
./gradlew bootRun

# Ejecutar tests
./gradlew test

# Limpiar y construir
./gradlew clean build

# Generar JAR ejecutable
./gradlew bootJar

# Ver dependencias
./gradlew dependencies

# Actualizar Gradle Wrapper
./gradlew wrapper --gradle-version 8.5
